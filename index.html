function zaloot() {

  var dotss = document.getElementById("dotss");

  var moreTextt = document.getElementById("moree");

  var btnTextt = document.getElementById("myBtnn");


  if (dotss.style.display === "none") {

    dotss.style.display = "inline";

    btnTextt.innerHTML = "Read more"; 

    moreTextt.style.display = "none";

  } else {

    dotss.style.display = "none";

    btnTextt.innerHTML = "Collapse"; 

    moreTextt.style.display = "inline";

  }

}
